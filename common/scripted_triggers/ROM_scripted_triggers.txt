#scripted trigger for Hungary

is_available_fighter_ROM = {
	OR = {
		AND = {
			has_tech = jet_fighter_1
			ROM = { 
				NOT = { has_tech = jet_fighter_1}
			}
		}
		AND = {
			has_tech = jet_fighter_2
			ROM = { 
				NOT = { has_tech = jet_fighter_2}
			}
		}
		AND = {
			has_tech = jet_fighter_3
			ROM = { 
				NOT = { has_tech = jet_fighter_3}
			}
		}
	}
	NOT = {
		has_war_with = ROM
	}
}
