country_event = {
	id = NE_SOV.8
	title = NE_SOV.8.t
	desc = NE_SOV.8.desc
	picture = GFX_report_event_generic_sign_treaty3

	is_triggered_only = yes

	# 수락
	option = {
		name = NE_SOV.8.a
		ai_chance = {
			factor = 50
		}
		effect_tooltip = {
			FIN = { add_popularity = { ideology = communism popularity = 0.2 } }
			#add_ideas = anti_NE_SOV_pact
			add_opinion_modifier = { target = SOV modifier = influential_sphere_by_SOV }
		}
		set_global_flag = SOV_flag_finland_conflict_resolution
		SOV = {
			country_event = { id = NE_SOV.9 }
		}
	}

	# 거부
	option = {
		name = NE_SOV.8.b
		ai_chance = {
			#factor = 50
			modifier = {
				factor = 0.5
				strength_ratio = { tag = SOV ratio < 1 }
			}
			modifier = {
				factor = 0.2
				strength_ratio = { tag = SOV ratio < 0.5 }
			}
			modifier = {
				factor = 0.5
				SOV = { strength_ratio = { tag = SOV ratio > 0.8 } }
			}
			modifier = {
				factor = 0
				SOV = { strength_ratio = { tag = SOV ratio > 0.8 } }
				strength_ratio = { tag = SOV ratio < 0.5 }
			}
		}
		set_global_flag = SOV_flag_finland_conflict_resolution
		complete_national_focus = FIN_strategy_decisions
		SOV = {
			country_event = { id = NE_SOV.10 }
		}
	}
}

# 중립화냐 나토화냐 (핀란드)
country_event = {
	id = NE_SOV.11
	title = NE_SOV.11.t
	desc = NE_SOV.11.desc
	picture = GFX_report_event_german_troops

	is_triggered_only = yes

	trigger = {
		NOT = { has_war_with = SOV }
	}
	
	#핀란드 - 중립화를 선택
	option = {
		name = NE_SOV.11.a
		ai_chance = {
			factor = 20
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		SOV = { country_event = { id = NE_SOV.12 } }
		effect_tooltip = {
			leave_faction = yes
			add_ideas = authoritarianism_idea
		}
		hidden_effect = {
			#news_event = { id = news.203 hours = 6 }
		}
	}
	# 핀란드 - 나토를 선택
	option = {
		name = NE_SOV.11.b
		ai_chance = {
			factor = 80
			modifier = {
				factor = 3
				strength_ratio = {
					tag = SOV
					ratio > 0.7
				}
			}
			modifier = {
				factor = 0.1
				any_other_country = {
					is_major = yes
					is_in_faction_with = FIN
				}
			}
		}
		SOV = { country_event = { id = NE_SOV.13 } 
			if = {
				limit = {
					is_in_faction_with = FIN
					is_faction_leader = yes
				}
				remove_from_faction = FIN
			}
			if = {
				limit = { 
					is_in_faction_with = FIN
					is_faction_leader = no
				}
				random_other_country = {
					limit = { 
						is_in_faction_with = SOV
						is_faction_leader = yes
					}
					remove_from_faction = SOV
				}
			}
		}		
		hidden_effect = {
			#news_event = { hours = 1 id = news.204 }
		}
		effect_tooltip = {
			SOV = {
				create_wargoal = {
				type = topple_government
				target = FIN
				}
			}
		}
	}
}
